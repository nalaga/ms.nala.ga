using Microsoft.Win32;
using System;
using System.Collections.Generic;
using System.ComponentModel;

using System.Text;
using System.Threading.Tasks;

namespace WpfApp1
{
    /// <summary>
    /// MainWindow.xaml에 대한 상호 작용 논리
    /// </summary>
    public partial class MainWindow : Window
    {
        public MainWindow()
        {
            InitializeComponent();
        }
 
        List<Man> items = new List<Man>();
        private void btnADD_Click(object sender, RoutedEventArgs e)
        {
            items.Add(new Man() { Name = “111”, Age = 23 });
            items.Add(new Man() { Name = “222”, Age = 23 });
            items.Add(new Man() { Name = “333”, Age = 23 });
            items.Add(new Man() { Name = “444”, Age = 23 });
            items.Add(new Man() { Name = “555”, Age = 23 });
            ListView01.ItemsSource = items;
            ListView01.Items.Refresh();
        }
 
        private void btn_DEL_Click(object sender, RoutedEventArgs e)
        {
            foreach (Man item in ListView01.SelectedItems)
            {
                items.Remove(item);
            }
            ListView01.Items.Refresh();
        }
    }
 
    public class Man
    {
        public string Name { get; set; }
        public int Age { get; set; }
    }
}
