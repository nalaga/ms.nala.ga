  1. – 데이터의 정확성을 보장하기 위함
  2. insert into null_testvalues(2, ”);
  3. 테이블 구조변경을 통해 name 컬럼을 not null로 변경지정해 본다.

무결성 제약조건 integrity constraint

– 데이터의 정확성을 보장하기 위함

1. not null

2. check

3. unique (유일성)

3. primary key (유일성보장+동일성),

5. foreign key

ex ) 회원가입시 *표시로 필수입력항목,, 채우지 않고 가입하려면 빠꾸당함 왜냐면 필수입력항목이 not null로 설계되어있기 때문임

1. not null

create table null_test(

id      number(4)       ,

name    varchar2(30)

);

이건 두개다 null 허용하는건데

create table null_test(

id      number(4) not null   ,

name    varchar2(30)

);

이렇게 하면 not null이 됨

확인

desc null_test

이제 값을 넣어보자

insert into null_test

values(1, ‘에즈라’);

select * from null_test;

근데

insert into null_test

values(null, ‘에즈라’);

오류발생

notnull이기 때문

insert into null_test

values(2, ”);

근데 이건됨 name 값은 not null 아니기 때문

테이블 구조변경을 통해 name 컬럼을 not null로 변경지정해 본다.

alter table null_test

modify (name not null);

오류남 왜?? 이미 null 값으로 넣어놓은게 있어서

그럼 데이터변경해보자

update null_test set name=’제시’

where id=2;

그다음 다시,

alter table null_test

modify (name not null);

desc null_test

이제 name까지 not null로 바꼈다!

모든 제약조건은 관리가 되어있다.

제약조건 확인 

table 관련 제약조건 확인 user_constraints 라는 테이블

column 관련 제약조건 확인 – user_cons_columns 라는 테이블

*user_constraints 

select constraint_name, constraint_type

from user_constraints

where table_name = ‘NULL_TEST’;

user_constraints에는

hr유저가 가지고있는 모든 constraints 를 볼수있는데

이중 null_test에 대한 constraints만 보고싶어서 where

*이때 null_test는 대문자로 써야함

*user_cons_columns 

select constraint_name, column_name

from user_cons_columns

where table_name=’NULL_TEST’;

이제, 

NULL_TEST 테이블의 제약조건명, 컬럼명, 제약조건유형을 조회해보자

select uc.constraint_name, cc.column_name, uc.constraint_type

from user_constraints uc , user_cons_columns cc

where  uc.constraint_name=cc.constraint_name and uc.table_name=’NULL_TEST’;

*table_name 앞에도 어떤 테이블의 것인지 명시해줘야 함.

만약 테이블조인을 안하고싶다면

어느테이블에 관점을 맞출거냐가 관점임

select constraint_name, column_name,

    (select constraint_type from user_constraints where c.constraint_name=constraint_name) constraint_type

    from user_cons_columns c

    where table_name=’NULL_TEST’;

이렇게 해도 똑같은 결과.

2. check

특정 조건에 맞는 데이터만 저장되게 하고자 하는 경우

gender 에는 남성,여성 이라는 데이터만 저장할수 있게 조건주고자 한다.

create table check_test2(

        name varchar2(30) not null,

        gender varchar2(6) not null check(gender in (‘남성’,’여성’)),

        salary number(6),

        dept_id number(4), 

        constraint check_test_salary_ck check(salary>=3000));

* constraint의 이름을 저렇게 명명함으로서 어떤 테이블의 어떤 컬럼의 constraint인지 알 수 있게함

select constraint_name, column_name,

    (select constraint_type from user_constraints where c.constraint_name=constraint_name) constraint_type

    from user_cons_columns c

    where table_name=’CHECK_TEST2′;

constraint인 check_test2_salary_ck 가 추가된걸 알수있고,

이제 데이터 집어넣어보자

insert into check_test2

values (‘에즈라’,’남성’,3000,null);

이건되는데,

insert into check_test2

values (‘시아0’,’여자‘,5000,null);

이건 오류

왜?

남성 또는 여성으로만 저장되도록 check 했기때문,,

insert into check_test2

values (‘시아’,’여성‘,5000,null);

이렇게 하면 됨

또,

update check_test2 set salary = 3500 where name=’에즈라’;

이건되는데

update check_test2 set salary = 2000 where name=’시아’;

이건안됨

왜?

데이터무결성위한 check constraint에 3000이상 연봉만 저장되게 해놨기 때문

update check_test2 set salary = 6000 where name=’시아’;

select * from check_test2;

완성!

* constraint 추가 / 삭제

check_test 테이블에 지정된 constraint를 삭제해보자

즉

테이블의 구조를 변경(alter table)해야 함

alter table check_test2

drop constraint check_test_salary_ck;

*drop coloumn 대신 drop constraint 한것

이제 salary 2000으로 해도 오류안남

다시,

check_test2 테이블에 새로운 constraint 추가 하려면

alter table check_test2

add constraint check_test2_salary_ck 

        check(salary between 2000 and 8000);

3. unique

유일성 보장받아야 하는 경우,

즉 똑같은 데이터가 저장되지 않도록 할 때 지정하는것! – null 허용

create table unique_test(

id   number(4) not null,

name    varchar2(30) not null,

email   varchar2(30),         /*constraint unique_test_email_nk unique,*/

phone   varchar2(13),

constraint unique_test_email_uk unique(email)

);

*보라색은 컬럼레벨에서 선언한거고, 파란색처럼 해도 되고 보라색처럼 선언해도 되고 상관 無

insert into unique_test

values (1,’에즈라’, ‘[email protected]’,null);

insert into unique_test

values (2,’제시’, ‘[email protected]’,null);

제시는 에즈라와 이메일이 같아서 오류 ㅠ 

하지만

insert into unique_test

values (2,’제시’, ”,null);

unique는 null 허용! null 은 됨

* 복합 constraint (=복합 key)

여러개의 컬럼으로 하나의 constraint 지정 가능

unique_test 테이블에 unique constraint 추가 해 보자

alter table unique_test

add constraint unique_test_nameemail_uk  unique(name, phone);

폰번호과 이름을 엮어서, 두개의 데이터를 함꼐 볼 때, 유일성을 보장받기만 하면됨

이름만 다르면 전번같아도 유일성보장

이름같더라도 전번다르면 유일성 보장

4. primary key

 어떤 테이블에서 데이터 행을 구별하기 위한 식별자

데이터 값이 반드시 있어야 함 not null

+

데이터 값이 똑같은데 있으면 안됨 unique

        

    ex ) 주민번호, 학번

        똑같은 주번있을수 없고 주번없는사람이 없다.

        

create table dept_tast(

dept_id     number(4) /*constraint dept_test_id_pk primary key*/,

dept_name       varchar2(50),

constraint dept_test_id_pk primary key (dept_id)

);

부서코드에 pk를 주고,

insert into dept_tast

values (10,’영업부’);

insert into dept_tast

values (20, ‘총무부’);

값 넣고,

update dept_tast set dept_id=10

where dept_id=20;

부서코드 20인 총무부의 부서코드를 10으로 바꾸려 했으나 오류남

primary key이기 때문임

5. foreign key

(=fk, 외래키, 참조키)

부모자식간의 관계를 갖게되는 형태

부모테이블에 있는 데이터 값만 저장되도록 하고자 할때 지정함

ex ) 회원가입 한 사람만 게시판에 글 쓸 수 있게 할 때

* 자식테이블에 foreign key 를 지정하고, null 을 허용함

* foreign key 를 지정하려면 반드시 부모테이블에서 pk에 해당하는 컬럼을 참조(references) 해야 한다.

create table emp_test(

    emp_id  number(4) constraint emp_test_id_pk primary key,

    name    varchar(30),

    dept_id number(4)     /*foreign key만 쓴다고 지정되는게 아님 레퍼런스해야함*/ /* constraint emp_test_dept_fk references departments(department_id)*/,

    job_id  varchar(30),

    constraint emp_test_dept_fk foreign key (dept_id) references departments(department_id)

    );

 insert into emp_test

    values (100, ‘홍길동’, 10, ‘ST_CLERK’);

이건 삽입되는데,

insert into emp_test

    values (101, ‘전우치’, 500, ‘ST_CLERK’);

이건안됨 ! 왜? 부모 table인 departments에 500번 dept_id가 없기 때문!

    insert into emp_test

     values (101, ‘전우치’,50, ‘ST_CLERK’);

이렇게 하면 된다.

    select * from emp_test;

*foreign key를 추가하기

  emp_test 테이블의 job_id 컬럼에

    jobs 테이블의 job_id 컬럼을 참조하도록 fk를 지정

alter table emp_test

    add constraint emp_test_job_fk foreign key (job_id)

        references jobs(job_id);
